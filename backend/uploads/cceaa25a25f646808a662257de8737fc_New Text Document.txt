from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session

from ..models import Doctor
from ..schemas import DoctorCreate, DoctorResponse
from ..dependencies import get_db, admin_required, get_current_user

router = APIRouter(prefix="/doctors", tags=["Doctors"])


# ================= CREATE DOCTOR =================
@router.post("", response_model=DoctorResponse, dependencies=[Depends(admin_required)])
def create_doctor(data: DoctorCreate, db: Session = Depends(get_db)):

    # ✅ Check duplicate email
    existing = db.query(Doctor).filter(Doctor.email == data.email).first()
    if existing:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Doctor already exists"
        )

    doctor = Doctor(**data.dict())

    db.add(doctor)
    db.commit()
    db.refresh(doctor)

    return doctor


# ================= GET ALL DOCTORS =================
@router.get("", response_model=list[DoctorResponse])
def get_doctors(
    db: Session = Depends(get_db),
    user=Depends(get_current_user)
):
    return db.query(Doctor).filter(Doctor.is_active == True).all()


# ================= GET SINGLE DOCTOR =================
@router.get("/{doctor_id}", response_model=DoctorResponse)
def get_doctor(
    doctor_id: int,
    db: Session = Depends(get_db),
    user=Depends(get_current_user)
):
    doctor = db.query(Doctor).filter(Doctor.id == doctor_id).first()

    if not doctor:
        raise HTTPException(status_code=404, detail="Doctor not found")

    return doctor


# ================= UPDATE DOCTOR =================
@router.put("/{doctor_id}", response_model=DoctorResponse, dependencies=[Depends(admin_required)])
def update_doctor(
    doctor_id: int,
    data: DoctorCreate,
    db: Session = Depends(get_db)
):
    doctor = db.query(Doctor).filter(Doctor.id == doctor_id).first()

    if not doctor:
        raise HTTPException(status_code=404, detail="Doctor not found")

    for key, value in data.dict().items():
        setattr(doctor, key, value)

    db.commit()
    db.refresh(doctor)

    return doctor


# ================= DELETE (SOFT DELETE) =================
@router.delete("/{doctor_id}", dependencies=[Depends(admin_required)])
def delete_doctor(
    doctor_id: int,
    db: Session = Depends(get_db)
):
    doctor = db.query(Doctor).filter(Doctor.id == doctor_id).first()

    if not doctor:
        raise HTTPException(status_code=404, detail="Doctor not found")

    doctor.is_active = False

    db.commit()

    return {"message": "Doctor soft deleted"}
